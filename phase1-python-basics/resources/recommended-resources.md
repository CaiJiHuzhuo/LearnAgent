## 📋 常用速查表（Cheat Sheet）

### Python 基础速查

```python
# 数据类型
type(x)          # 查看类型
isinstance(x, T) # 类型检查
int("42")        # 类型转换
str(42)          # 类型转换
list(range(10))  # 生成列表
dict.fromkeys(keys, default)  # 创建字典

# 字符串
f"Hello {name}"  # f-string 格式化
"hello".upper()  # 大写
"  hello  ".strip()  # 去空白
"a,b,c".split(",")  # 分割
",".join(["a", "b"])  # 合并

# 列表
lst.append(x)         # 末尾添加
lst.extend(other)     # 合并列表
lst.insert(i, x)      # 指定位置插入
lst.remove(x)         # 删除元素
lst.pop()             # 删除并返回最后一个
lst.sort(key=..., reverse=...)  # 排序
sorted(lst)           # 返回新排序列表
[x for x in lst if cond]  # 列表推导式

# 字典
d.get(key, default)   # 安全获取
d.setdefault(key, []) # 设置默认值
d.items()             # 键值对视图
d.keys() / d.values() # 键/值视图
d | other             # 合并字典（3.9+）
{k: v for k, v in d.items() if cond}  # 字典推导式

# 集合
s.add(x)       # 添加
s.remove(x)    # 删除（不存在抛异常）
s.discard(x)   # 删除（不存在不报错）
s | t          # 并集
s & t          # 交集
s - t          # 差集
```

### Python 进阶速查

```python
# 装饰器
import functools
def decorator(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        # 前置逻辑
        result = func(*args, **kwargs)
        # 后置逻辑
        return result
    return wrapper

# 上下文管理器
from contextlib import contextmanager
@contextmanager
def managed():
    # 初始化
    yield resource
    # 清理

# 生成器
def gen():
    yield 1
    yield 2
    yield 3

# 类型提示
from typing import Optional, Union, Callable
def func(x: int, y: Optional[str] = None) -> list[int]: ...

# Pydantic
from pydantic import BaseModel, Field
class Model(BaseModel):
    field: str = Field(min_length=1)
    optional: int | None = None
    
model = Model(field="test")
model.model_dump()         # 转字典
model.model_dump_json()    # 转 JSON
Model.model_validate(dict) # 从字典创建
```

### asyncio 速查

```python
import asyncio

# 基础
async def coro(): ...          # 协程函数
await coro()                   # 等待协程
asyncio.run(coro())            # 运行协程（入口点）

# 并发
results = await asyncio.gather(coro1(), coro2(), coro3())
results = await asyncio.gather(*coros, return_exceptions=True)

# Task
task = asyncio.create_task(coro())  # 创建任务（立即调度）
result = await task                  # 等待任务完成
task.cancel()                        # 取消任务

# 超时
try:
    result = await asyncio.wait_for(coro(), timeout=5.0)
except asyncio.TimeoutError:
    pass

# 并发限制
sem = asyncio.Semaphore(10)
async with sem:
    await coro()
```

### FastAPI 速查

```python
from fastapi import FastAPI, Depends, HTTPException
from pydantic import BaseModel

app = FastAPI()

# 路由
@app.get("/path")
@app.post("/path")
@app.put("/path/{id}")
@app.patch("/path/{id}")
@app.delete("/path/{id}")

# 请求数据
# 路径参数：在路由中 {param}，函数参数中 param: type
# 查询参数：函数参数中 param: type = default
# 请求体：函数参数中 body: PydanticModel

# 响应
return {"key": "value"}          # 返回字典
return data                       # 返回 Pydantic 模型实例

# 错误
raise HTTPException(status_code=404, detail="Not found")

# 依赖注入
def get_db(): yield db
@app.get("/data")
async def endpoint(db: Depends(get_db)): ...
```
